// ── 컨테이너 ──
const wrap = dv.el("div", "");
wrap.style.cssText = `
  display: flex; flex-direction: column;
  align-items: center; padding: 60px 20px; gap: 20px;
`;

// ── 제목 ──
const title = dv.el("div", "📡 긱뉴스 수집기");
title.style.cssText = `
  font-size: 1.8em; font-weight: bold; color: #58a6ff;
  letter-spacing: 3px; margin-bottom: 10px;
`;
wrap.appendChild(title);

// ── 날짜 선택기 ──
const today = new Date().toISOString().split("T")[0];

const dateLabel = dv.el("div", "수집 날짜");
dateLabel.style.cssText = `color: #888; font-size: 0.85em; margin-bottom: 4px;`;
wrap.appendChild(dateLabel);

const datePicker = document.createElement("input");
datePicker.type = "date";
datePicker.value = today;
datePicker.style.cssText = `
  background: #161b22; color: #c9d1d9;
  border: 1px solid #30363d; border-radius: 8px;
  padding: 10px 20px; font-size: 1.1em;
  text-align: center; cursor: pointer;
  margin-bottom: 10px;
`;
wrap.appendChild(datePicker);

// ── 버튼 ──
const btn = document.createElement("button");
btn.textContent = "▶  수집 시작";
btn.style.cssText = `
  background: linear-gradient(135deg, #0f3460, #1a5276);
  color: #fff; border: 2px solid #4a9eff;
  padding: 18px 90px; border-radius: 12px;
  font-size: 1.3em; font-weight: bold; cursor: pointer;
  letter-spacing: 2px; min-width: 320px;
  box-shadow: 0 0 20px rgba(74,158,255,0.3);
`;
btn.onmouseover = () => btn.style.background = "linear-gradient(135deg, #1a5276, #2471a3)";
btn.onmouseout  = () => btn.style.background = "linear-gradient(135deg, #0f3460, #1a5276)";
wrap.appendChild(btn);

// ── 상태 텍스트 ──
const status = dv.el("div", "긱뉴스 | AI 제외 | 내용있는 글 최대 20개");
status.style.cssText = `color: #555; font-size: 0.85em;`;
wrap.appendChild(status);

// ── 클릭 이벤트 (경로를 본인 환경에 맞게 수정) ──
btn.onclick = () => {
  const dateVal = datePicker.value;
  if (!dateVal) { new Notice("날짜를 선택해 주세요."); return; }

  try {
    const { exec } = require("child_process");
    const python = "d:\\내프로젝트폴더\\venv\\Scripts\\python.exe"; // ← 수정
    const script = "d:\\내프로젝트폴더\\geek\\scripts\\collector.py"; // ← 수정
    const cmd    = `"${python}" "${script}" ${dateVal}`;

    btn.disabled = true;
    btn.style.opacity = "0.5";
    status.textContent = `⏳ ${dateVal} 수집 중...`;
    status.style.color = "#4a9eff";

    exec(cmd, { encoding: "buffer" }, (err, stdout, stderr) => {
      btn.disabled = false;
      btn.style.opacity = "1";

      if (err) {
        const errMsg = stderr.toString("utf8") || err.message;
        status.textContent = "❌ 오류 발생";
        status.style.color = "#ff5555";
        new Notice("오류: " + errMsg.slice(0, 150), 8000);
      } else {
        const parts = dateVal.split("-");
        const folderName = parts[0].slice(2) + "-" + parts[1] + "-" + parts[2];
        status.textContent = `✅ 완료! geek/${folderName}/ 폴더를 확인하세요.`;
        status.style.color = "#4caf50";
        new Notice(`✅ ${folderName} 수집 완료!`, 5000);
      }
    });

  } catch(e) {
    new Notice("실행 오류: " + e.message, 8000);
  }
};
